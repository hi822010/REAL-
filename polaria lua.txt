coroutine.wrap(function()
	local script = Instance.new("Script", Instances.rain_2)
	local button = script.Parent  -- The button this script is inside
	local isLooping = false       -- Toggle to control the loop
	local spawnDelay = 2          -- Time in seconds between each batch of balls
	local ballsPerBatch = 10      -- Number of balls to spawn in each batch
	
	local function spawnRandomFallingBall()
		-- Create the part (ball)
		local ball = Instance.new("Part")
		ball.Shape = Enum.PartType.Ball
		ball.Size = Vector3.new(5, 5, 5)  -- Set the size of the ball
		ball.Color = Color3.fromRGB(17, 17, 17)  -- "Really Black" color
		ball.Anchored = false  -- Allow it to fall
		ball.CanCollide = true
	
		-- Generate a random position within the map bounds
		local randomX = math.random(-250, 250)
		local randomZ = math.random(-250, 250)
		ball.Position = Vector3.new(randomX, 100, randomZ)  -- Random position in the sky
		ball.Parent = workspace  -- Add the ball to the workspace
	
		-- Add fire to the ball
		local fire = Instance.new("Fire")
		fire.Size = 10
		fire.Heat = 20
		fire.Parent = ball
	
		-- Create an explosion on impact
		ball.Touched:Connect(function(hit)
			if hit:IsA("Terrain") or hit.Anchored then
				-- Create an explosion when the ball hits the ground
				local explosion = Instance.new("Explosion")
				explosion.Position = ball.Position
				explosion.BlastRadius = 10
				explosion.BlastPressure = 50000
				explosion.Parent = workspace
	
				ball:Destroy()  -- Destroy the ball after explosion
			elseif hit.Parent:FindFirstChild("Humanoid") then
				-- Kill the player if the ball hits a character
				hit.Parent:FindFirstChild("Humanoid").Health = 0
			end
		end)
	end
	
	local function spawnBallBatch()
		for i = 1, ballsPerBatch do
			spawnRandomFallingBall()  -- Spawn multiple balls in a loop
		end
	end
	
	local function toggleBallLoop()
		isLooping = not isLooping  -- Toggle the looping state
		if isLooping then
			while isLooping do
				spawnBallBatch()
				wait(spawnDelay)  -- Delay before the next batch
			end
		end
	end
	
	-- Connect the button click to toggle the spawning loop
	button.MouseButton1Click:Connect(toggleBallLoop)
	
end)()

	local function fillWithWater()
		local terrain = workspace.Terrain -- Get the terrain object
		local sizeX = 1024
		local sizeZ = 1024
		local waterHeight = 10 -- Height of the water level
		local step = 4 -- Step size for the grid
	
		for x = -sizeX / 2, sizeX / 2, step do

				terrain:FillRegion(region, Enum.Material.Water, 4)
			end
		end
	end
	
	-- Listen for the event from the client and trigger the flood
	remoteEvent.OnServerEvent:Connect(fillWithWater)

coroutine.wrap(function()

	-- Variables
	local button = script.Parent
	local players = game:GetService("Players")
	local toggle = false -- Visibility toggle
	
	-- Function to toggle visibility
	local function toggleVisibility()
		toggle = not toggle -- Switch the toggle state
	
		for _, player in pairs(players:GetPlayers()) do
			if player.Character and player.Character:FindFirstChild("Humanoid") then
				for _, part in pairs(player.Character:GetDescendants()) do
					-- Only affect visible parts of the character
					if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
						part.Transparency = toggle and 1 or 0
						part.CanCollide = not toggle
					elseif part:IsA("Decal") then
						part.Transparency = toggle and 1 or 0 -- Hide decals like faces
					end
				end
			end
		end
	end
	
	-- Button click event
	button.Touched:Connect(function(hit)
		local character = hit.Parent
		if character:FindFirstChild("Humanoid") then
			toggleVisibility()
		end
	end)
	
end)()

coroutine.wrap(function()
	local script = Instance.new("Script", Instances.cina)
	local button = script.Parent -- Reference to the button
	local Players = game:GetService("Players")
	local Workspace = game:GetService("Workspace")
	
	-- Function to handle the aesthetics changes
	local function Aesthetics(player)
		local char = player.Character or player.CharacterAdded:Wait() -- Ensure character is loaded
		local face = char:FindFirstChild("Head"):FindFirstChild("face")
		if face then
			face.Texture = "http://www.roblox.com/asset/?id=111865733232046"
		end
	end
	
	-- Function to play the song
	local function playSong()
		local song = Instance.new("Sound")
		song.Volume = 0.5
		song.SoundId = "rbxassetid://123933240720384"
		song.Parent = Workspace
		song.Looped = false
		song:Play()
	end
	
	-- Function to apply decals to all WedgeParts in the workspace
	local function applyDecals()
		for _, v in pairs(Workspace:GetDescendants()) do
			if v:IsA("WedgePart") then
				local faces = {
					{Face = Enum.NormalId.Front, Texture = "http://www.roblox.com/asset/?id=124872714710239"},
					{Face = Enum.NormalId.Back, Texture = "http://www.roblox.com/asset/?id=111616848"},
					{Face = Enum.NormalId.Bottom, Texture = "http://www.roblox.com/asset/?id=111616664"},
					{Face = Enum.NormalId.Left, Texture = "http://www.roblox.com/asset/?id=112188192"},
					{Face = Enum.NormalId.Right, Texture = "http://www.roblox.com/asset/?id=111616722"},
					{Face = Enum.NormalId.Top, Texture = "http://www.roblox.com/asset/?id=111616450"}
				}
				for _, faceData in pairs(faces) do
					local decal = Instance.new("Decal")
					decal.Texture = faceData.Texture
					decal.Face = faceData.Face
					decal.Parent = v
				end
			end
		end
	end
	
	-- Button click handler
	button.MouseButton1Down:Connect(function()
		local player = Players:GetPlayerFromCharacter(script.Parent.Parent) -- Get the player from the button's parent (ensure this button is placed in a player's GUI)
	
		if player then
			-- Trigger Aesthetics function for the player
			Aesthetics(player)
	
			-- Play the song
			playSong()
	
			-- Apply decals to all WedgeParts in the workspace
			applyDecals()
		end
	end)
	
end)()

		gui.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
	
	gui.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = gui.Position
			
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	
	gui.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end)()

coroutine.wrap(function()
	local script = Instance.new("LocalScript", Instances.troll)
	local TweenService = game:GetService("TweenService")  -- Access TweenService
	local imageLabel = script.Parent  -- The ImageLabel this script is attached to
	local holdFrame = imageLabel:WaitForChild("hold")  -- The 'hold' frame inside the ImageLabel
	
	-- TweenInfo for rotating the ImageLabel
	local rotationTweenInfo = TweenInfo.new(
		1.5,  -- Duration for rotation (seconds)
		Enum.EasingStyle.Quad,  -- Easing style for smooth rotation
		Enum.EasingDirection.Out  -- Easing direction
	)
	
	-- TweenInfo for the 'hold' frame movement
	local frameTweenInfo = TweenInfo.new(
		0.5,  -- Duration for the size and position change (seconds)
		Enum.EasingStyle.Quad,  -- Easing style for smooth transition
		Enum.EasingDirection.Out  -- Easing direction
	)
	
	-- TweenInfo for the 'hold' frame background transparency
	local transparencyTweenInfo = TweenInfo.new(
		0.5,  -- Duration for the transparency change (seconds)
		Enum.EasingStyle.Quad,  -- Easing style for smooth transparency change
		Enum.EasingDirection.Out  -- Easing direction
	)
	
	-- Initial properties for ImageLabel and 'hold' frame
	local initialRotation = 0
	local initialSize = UDim2.new(0, 21, 0, 10)
	local initialPosition = UDim2.new(0.394, 0, -0.172, 0)
	local targetSize = UDim2.new(0, 211, 0, 79)
	local targetPosition = UDim2.new(-2.242, 0, 0.101, 0)  -- New target position
	
	-- Set initial properties
	holdFrame.Position = initialPosition
	holdFrame.Size = initialSize
	holdFrame.Visible = false  -- Ensure hold frame is initially invisible
	holdFrame.BackgroundTransparency = 1  -- Initial transparency (invisible)
	imageLabel.Rotation = initialRotation
	
	-- Create the tween for ImageLabel rotation (single rotation)
	local rotationTween = TweenService:Create(imageLabel, rotationTweenInfo, {Rotation = 360})
	
	-- Create the tween for the 'hold' frame movement
	local frameTween = TweenService:Create(holdFrame, frameTweenInfo, {Position = targetPosition, Size = targetSize})
	
	-- Create the tween for the 'hold' frame background transparency (fade in)
	local transparencyTween = TweenService:Create(holdFrame, transparencyTweenInfo, {BackgroundTransparency = 0})
	
	-- When the player hovers over the ImageLabel (MouseEnter)
	imageLabel.MouseEnter:Connect(function()
		-- Play the rotation tween for ImageLabel (one-time rotation)
		rotationTween:Play()
	
		-- Play the tween to change the 'hold' frame position and size
		frameTween:Play()
	
		-- Play the transparency tween to make the 'hold' frame visible smoothly
		holdFrame.Visible = true
		transparencyTween:Play()
	end)
	
	-- When the player stops hovering over the ImageLabel (MouseLeave)
	imageLabel.MouseLeave:Connect(function()
		-- Reset the rotation to initial position smoothly
		local resetRotationTween = TweenService:Create(imageLabel, rotationTweenInfo, {Rotation = 0})
		resetRotationTween:Play()
	
		-- Reset the 'hold' frame position and size to its initial state
		local resetFrameTween = TweenService:Create(holdFrame, frameTweenInfo, {Position = initialPosition, Size = initialSize})
		resetFrameTween:Play()
	
		-- Fade the 'hold' frame background to invisible (BackgroundTransparency = 1)
		local resetTransparencyTween = TweenService:Create(holdFrame, transparencyTweenInfo, {BackgroundTransparency = 1})
		resetTransparencyTween:Play()
	
		-- Make the 'hold' frame invisible again after the transparency tween completes
		resetTransparencyTween.Completed:Connect(function()
			holdFrame.Visible = false
		end)
	end)
end)()
end)()
